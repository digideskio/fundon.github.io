Install Go on Ubuntu

Installation

VERSION="1.3.3"
wget http://golang.org/dl/go$VERSION.linux-amd64.tar.gz
tar -C /usr/local -xzf go$VERSION.linux-amd64.tar.gz

export PATH="$PATH:/usr/local/go/bin"

#export PATH="$PATH:$GOPATH/bin"
